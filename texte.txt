function fresh() {
  return { anglers: ANGLERS.map(a => ({
    id:a.id,
    counts:{ pike:0, perch:0, zander:0, catfish:0 },
    sizes:{ pike:'', perch:'' },
    milestone3:{ pike:false, perch:false, zander:false, catfish:false },

  }))};

let S = fresh();
const save = () => { try { localStorage.setItem('pw4', JSON.stringify(S)); } catch(e){} };
const load = () => { try { const d=localStorage.getItem('pw4'); if(d) S=JSON.parse(d); } catch(e){} };

function biggestSize(spId) {
  const min = SPECIES.find(s=>s.id===spId).minSize;
  let best={ size:0, ts:Infinity, id:-1 };
  S.anglers.forEach(a => {
    const v = parseFloat(a.sizes[spId])||0;
    if(v<min||v===0) return;
    if(v>best.size||(v===best.size&&a.sizets[spId]<best.ts)) best={size:v,ts:a.sizets[spId],id:a.id};
  });
  return best.id;
}

function totalFish(a) { return Object.values(a.counts).reduce((s,v)=>s+v,0); }

function volumeLeader() {
  let max=0, winner=-1;
  S.anglers.forEach(a=>{ const t=totalFish(a); if(t>max){max=t;winner=a.id;} });
  return max>0?winner:-1;
}

function computeScores() {
  const pikeBest=biggestSize('pike'), perchBest=biggestSize('perch'), volWin=volumeLeader();
  return S.anglers.map(a => {
    let pts=0; const bd=[];
    SPECIES.forEach(sp => { const p=a.counts[sp.id]*sp.ptPer; if(p>0){pts+=p;bd.push(`${sp.emoji}×${a.counts[sp.id]}`);} });
    SPECIES.forEach(sp => { if(a.milestone3[sp.id]){pts+=1;bd.push(`×3${sp.emoji}`);} });
    if(a.id===pikeBest)  { pts+=1; bd.push('👑🐟'); }
    if(a.id===perchBest) { pts+=1; bd.push('👑🐠'); }
    if(a.id===volWin)    { pts+=1; bd.push('📊'); }
    if(a.bonus)          { pts+=1; bd.push('⭐'); }
    return { id:a.id, pts, bd };
  });
}

  save(); render();
  const el=document.getElementById(`v${aid}${spId}`);
  if(el){el.classList.remove('pop');void el.offsetWidth;el.classList.add('pop');}
}

function upSize(aid, spId, val) {
  const a = S.anglers[aid];
  const prev=parseFloat(a.sizes[spId])||0;
  const next=parseFloat(val)||0;
  a.sizes[spId]=val;
  if(next>prev) a.sizets[spId]=Date.now();
  save();
  const sp=SPECIES.find(s=>s.id===spId);
  const mailed=next>=sp.minSize&&next>0;
  const bc=val?(mailed?'rgba(126,207,126,.45)':'rgba(192,67,26,.5)'):'rgba(255,255,255,.08)';
  const inp=document.getElementById(`sz${aid}${spId}`);
  if(inp) inp.style.borderColor=bc;
  const pikeBest=biggestSize('pike'), perchBest=biggestSize('perch');
  const isWin=aid===(spId==='pike'?pikeBest:perchBest)&&mailed;
  const cr=document.getElementById(`cr${aid}${spId}`);
  if(cr) cr.textContent=isWin?'👑':'cm';
  renderStandings();
}

function uploadPhoto(aid) {
  const inp = document.getElementById(`photoInp${aid}`);
  const file = inp.files[0]; if(!file) return;
  const reader = new FileReader();
  reader.onload = function(ev){
    const canvas=document.createElement('canvas');
    const img=new Image();
    img.onload=function(){
      const size=120, scale=Math.min(size/img.width,size/img.height);
      canvas.width=img.width*scale; canvas.height=img.height*scale;
      canvas.getContext('2d').drawImage(img,0,0,canvas.width,canvas.height);
      S.anglers[aid].photo=canvas.toDataURL('image/jpeg',0.8);
      save(); render();
    };
    img.src=ev.target.result;
  };
  reader.readAsDataURL(file);
}

function toggleBonus(aid) { S.anglers[aid].bonus=!S.anglers[aid].bonus; save(); render(); }
function upBonusNote(aid,val) { S.anglers[aid].bonusNote=val; save(); }
function resetAll() { if(confirm('Reset the tournament?')) { S=fresh(); save(); render(); } }

function renderStandings() {
  const scores=computeScores();
  const maxPts=Math.max(...scores.map(s=>s.pts),1);
  const sorted=[...scores].sort((a,b)=>b.pts-a.pts);
  const rankEmoji=['🥇','🥈','🥉'], rankCls=['r1','r2','r3'];
  document.getElementById('standings').innerHTML=sorted.map((s,i)=>{
    const cfg=ANGLERS[s.id];
    const a=S.anglers[s.id];
    const thumb=a.photo
      ? `<img src="${a.photo}" style="width:28px;height:28px;border-radius:6px;object-fit:cover;border:1.5px solid ${cfg.hex}66;margin-right:6px;vertical-align:middle;"/>`
      : `<span style="font-size:1.1rem;margin-right:6px;">${cfg.emoji}</span>`;
    return `<div class="standings-row${i===0&&s.pts>0?' top':''}">
      <div class="st-rank ${rankCls[i]}">${rankEmoji[i]}</div>
      <div class="st-info">
        <div class="st-name" style="color:${cfg.hex}">${thumb}${cfg.name}</div>
        <div class="st-breakdown">${s.bd.join(' · ')||'—'}</div>
      </div>
      <div class="st-bar-wrap"><div class="st-bar" style="width:${Math.round(s.pts/maxPts*100)}%;background:${cfg.hex}"></div></div>
      <div class="st-pts" style="color:${cfg.hex}">${s.pts}</div>
    </div>`;
  }).join('');
}

function render() {
  const scores=computeScores();
  const topPts=Math.max(...scores.map(s=>s.pts));
  const leadId=topPts>0?scores.find(s=>s.pts===topPts).id:-1;
  const pikeBest=biggestSize('pike'), perchBest=biggestSize('perch');

  document.getElementById('dateBadge').textContent =
    '📅 '+new Date().toLocaleDateString('fr-FR',{weekday:'long',day:'numeric',month:'long',year:'numeric'}).toUpperCase();

  renderStandings();

  // CARDS
  document.getElementById('cards').innerHTML = S.anglers.map(angler => {
    const cfg=ANGLERS[angler.id];
    const sc=scores[angler.id];
    const isLead=angler.id===leadId&&sc.pts>0;

    const spRows = SPECIES.map(sp => {
      const count=angler.counts[sp.id];
      const ms=angler.milestone3[sp.id];
      let sizeCell='';
      if(sp.hasSize){
        const sv=angler.sizes[sp.id];
        const nv=parseFloat(sv)||0;
        const mailed=nv>=sp.minSize&&nv>0;
        const isWin=angler.id===(sp.id==='pike'?pikeBest:perchBest)&&mailed;
        const bc=sv?(mailed?'rgba(126,207,126,.45)':'rgba(192,67,26,.5)'):'rgba(255,255,255,.08)';
        sizeCell=`<div class="size-wrap">
          <input type="number" inputmode="decimal" class="size-inp" id="sz${angler.id}${sp.id}"
            placeholder="${sp.minSize}cm" value="${sv}" style="border-color:${bc}"
            oninput="upSize(${angler.id},'${sp.id}',this.value)"/>
          <span id="cr${angler.id}${sp.id}" class="${isWin?'crown-icon':'size-unit'}">${isWin?'👑':'cm'}</span>
        </div>`;
      } else {
        sizeCell=`<div style="width:50px"></div>`;
      }
      const dpts=(count*sp.ptPer)+(ms?1:0);
      return `<tr>
        <td><div class="sp-name"><span class="sp-emoji">${sp.emoji}</span>${sp.label}${ms?'<span class="milestone-pill">×3 ✓</span>':''}</div></td>
        <td>${sizeCell}</td>
        <td><div class="ctr">
          <button class="ctr-btn minus" onclick="upCount(${angler.id},'${sp.id}',-1)">−</button>
          <span class="ctr-val" id="v${angler.id}${sp.id}" style="color:${cfg.hex}">${count}</span>
          <button class="ctr-btn plus" onclick="upCount(${angler.id},'${sp.id}',1)">+</button>
        </div></td>
        <td class="pts-cell"><span class="pts-pill ${dpts>0?'on':'off'}">${dpts>0?'+'+dpts:'·'}</span></td>
      </tr>`;
    }).join('');

    const avatarHtml = angler.photo
      ? `<label class="avatar-upload" title="Changer la photo">
          <img class="avatar-img" src="${angler.photo}" style="border-color:${cfg.hex}66"/>
          <span class="avatar-edit-hint">✎</span>
          <input type="file" id="photoInp${angler.id}" accept="image/*" onchange="uploadPhoto(${angler.id})"/>
        </label>`
      : `<label class="avatar-upload" title="Ajouter une photo">
          <div class="avatar-placeholder" style="border-color:${cfg.hex}44">
            ${cfg.emoji}
          </div>
          <span class="avatar-edit-hint">📷</span>
          <input type="file" id="photoInp${angler.id}" accept="image/*" onchange="uploadPhoto(${angler.id})"/>
        </label>`;

    return `<div class="card${isLead?' leader':''}">
      <div class="card-accent" style="background:${cfg.hex}"></div>
      ${isLead?'<div class="leader-crown">👑</div>':''}
      <div class="card-head">
        ${avatarHtml}
        <div class="card-meta">
          <div class="card-name" style="color:${cfg.hex}">${cfg.name}</div>
          <div class="card-number">COMPETITOR ${cfg.number}</div>
        </div>
        <div class="card-score">
          <div class="card-pts" style="color:${cfg.hex}">${sc.pts}</div>
          <div class="card-pts-label">POINTS</div>
        </div>
      </div>
      <table class="species-table">
        <thead><tr><th>ESPÈCE</th><th>TAILLE</th><th style="text-align:center">CAPTURES</th><th style="text-align:center">PTS</th></tr></thead>
        <tbody>${spRows}</tbody>
      </table>
    </div>`;
  }).join('');

  // BONUS
  document.getElementById('bonusGrid').innerHTML=S.anglers.map(angler=>{
    const cfg=ANGLERS[angler.id];
    return `<div class="bonus-row">
      <div class="bonus-name" style="color:${cfg.hex}">${cfg.name}</div>
      <input class="bonus-inp" type="text" placeholder="Décrire l'action insolite…"
        value="${angler.bonusNote}" oninput="upBonusNote(${angler.id},this.value)"/>
      <button class="bonus-btn${angler.bonus?' on':''}" onclick="toggleBonus(${angler.id})">⭐</button>
    </div>`;
  }).join('');
}
